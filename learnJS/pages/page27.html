1)function camelize(str) {
return str
.split('-')
.map((word, i) => i == 0 ? word : word[0].toUpperCase() + word.slice(1))
.join('');
}
2)function filterRange(array, a, b) {
return array.filter(item => (a <= item && item <= b));
}
3)function filterRangeInPlace(arr, a, b) {
let val;
for (let i = 0; i < arr.length; i++) {
val = arr[i];
if (val < a || val > b) {
arr.splice(i, 1);
i--;
}
}
}

7)function sortByAge(arr) {
arr.sort((a, b) => a.age > b.age ? 1 : -1);
}
8)function shuffle(array) {
array.sort(() => Math.random() - 0.5);
}
9)function getAverageAge(users) {
return users.reduce((prev, user) => prev + user.age, 0) / users.length;
}
10)function unique(arr) {
let result = [];
for (let str of arr) {
if (!result.includes(str)) {
result.push(str);
}
}
return result;
}
